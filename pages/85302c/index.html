02.自定义组件

<!DOCTYPE html>
<html lang="en">
	<head>
		<meta charset="UTF-8" />
		<meta name="viewport" content="width=device-width, initial-scale=1.0" />
		<title>Document</title>
		<style>
			my-element {
				--header-bg: greenyellow;
			}
		</style>
	</head>
	<body class="dark">
		<template id="tmpl">
			<style>
				header {
					/* 通过外部变量改变样式 */
					background-color: var(--header-bg);
				}
				/* 选择 shadow 宿主 */
				:host {
					display: inline-block;
					width: 300px;
					background-color: #eee;
					padding: 10px;
				}

				/* 仅在 shadow 宿主与 selector 匹配时才应用样式 */
				:host([bold]) {
					font-weight: bold;
				}

				/* 仅当外部文档中的 shadow 宿主或它的任何祖先节点与 selector 匹配时才应用样式 */
				:host-context(.dark) {
					background-color: #333;
					color: #fff;
				}
			</style>
			<div>
				<header>
					<h2>TODO LIST</h2>
					<span></span>
				</header>
				<ul>
					<!-- 具名插槽 -->
					<slot name="item"></slot>
				</ul>
				<button>统计</button>
				<!-- 插槽默认 -->
				<slot name="footer">0</slot>
			</div>
		</template>
		<script>
			class MyElement extends HTMLElement {
				constructor() {
					super();
				}

				render() {
					let date = new Date(this.getAttribute('datetime') || Date.now());

					const span = this.shadow.querySelector('span');
					span.textContent = date.toLocaleString();
				}

				// 在元素被添加到文档之后，浏览器会调用这个方法
				//（如果一个元素被反复添加到文档／移除文档，那么这个方法会被多次调用）
				connectedCallback() {
					//「open」 —— shadow root 可以通过 elem.shadowRoot 访问
					//「closed」 —— elem.shadowRoot 永远是 null
					this.shadow = this.attachShadow({ mode: 'open' });
					this.template = tmpl.content.cloneNode(true);
					this.shadow.append(this.template);
					this.shadow.querySelector('button').addEventListener('click', () => {
						// slot.assignedElements({flatten: true/false}) 分配给插槽的 DOM 元素
						// slot.assignedNodes({flatten: true/false}) – 分配给插槽的 DOM 节点
						// node.assignedSlot – 返回 node 分配给的 <slot> 元素
						const itemsLength = this.shadow.querySelector('slot[name="item"]').assignedElements().length;
						console.log(itemsLength);
						this.shadow.querySelector('slot[name="footer"]').innerHTML = `${itemsLength}`;
					});

					this.render();
				}

				// 在元素从文档移除的时候，浏览器会调用这个方法
				// （如果一个元素被反复添加到文档／移除文档，那么这个方法会被多次调用）
				disconnectedCallback() {}

				// 属性数组，这些属性的变化会被监视
				static get observedAttributes() {
					return ['datetime'];
				}

				// 当上面数组中的属性发生变化的时候，这个方法会被调用
				attributeChangedCallback(name, oldValue, newValue) {
					this.render();
				}

				// 在元素被移动到新的文档的时候，这个方法会被调用
				// （document.adoptNode 会用到, 非常少见）
				adoptedCallback() {}
			}

			customElements.define('my-element', MyElement);

			setInterval(() => elem.setAttribute('datetime', new Date()), 1000);

			document.addEventListener('click', (e) => {
				// 当事件在组件外部捕获时，shadow DOM 中发生的事件将会以 host 元素作为目标 如 点击 button 元素
				// 如果事件发生在 slotted 元素上，实际存在于 light DOM 上，则不会发生重定向 如 点击 li 元素
				console.log(e.target, e.composedPath(), 'document');
			});
		</script>
		<my-element id="elem" bold>
			<li slot="item">读书</li>
			<li slot="item">看报</li>
			<li slot="item">打游戏</li>
		</my-element>
	</body>
</html>
