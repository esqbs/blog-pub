# 使用ServiceWorker离线缓存文件

Service Worker 是浏览器和网络之间的虚拟代理。

Service Worker 运行在一个与页面 JavaScript 主线程独立的线程上，并且无权访问 DOM 结构。

Service worker 可以控制网络请求、修改网络请求、返回缓存的自定义响应，或者合成响应。

// index.js
// 1. 首先需要在程序主入口中注册Service Worker 
if('serviceWorker' in navigator) {
  navigator.serviceWorker.register('/blog-pub/sw.js', {scope: './'})
};

// sw.js
// 缓存版本，当应用有了一个新版本时，可修改此版本号
let cachesName = "sw_v4"
// 初始化缓存的文件
let cachesList = [
  "/index.html"
]

// 2. 监听install事件，初始化缓存并添加离线应用时所需的文件
self.addEventListener('install', function(e) {
  e.waitUntil( // Service Worker 会等到 waitUntil 里面的代码执行完毕之后才开始安装。
    caches.open(cachesName) // 返回匹配 cacheName 的 Cache 对象，如果指定的 Cache 不存在，则使用该 cacheName 创建一个新的cache
    .then((cache) => cache.addAll(cachesList)) // 添加离线应用时所需的文件
    .then(self.skipWaiting()) // 强制waiting的service worker变成active的service worker
  )
})

// 3. 当缓存存在时，使用缓存来提供服务，而不重新请求数据。当请求的文件不在缓存中时，在响应之前将数据添加到缓存中。
self.addEventListener('fetch', function(e) {
  if(e.request.url.startsWith(self.location.origin)) { // 同域请求
    e.respondWith( // 响应控制
      caches.match(e.request) // 是否匹配缓存
      .then(function(r) { // 返回resolve为匹配 Response 的 Promise 对象。如果没有与指定request相匹配response，promise将使用 undefined resolve
        return r || fetch(e.request).then(function(response) { // 没匹配发起请求
          return caches.open(cachesName).then(function(cache) {
            cache.put(e.request, response.clone()) // 缓存请求的克隆数据
            return response // 返回数据
          })
        })
      })
    )
  }
})
// 4. service worker激活时，将之前的service worker删除，并让其他使用了之前的service worker的client也更新
self.addEventListener('activate', function(e) {
  let cacheWhitelist = [cachesName];
  e.waitUntil(
    caches.keys()
    .then(function(keyList) {
      return Promise.all(keyList.map(function(key) {
        if(cacheWhitelist.indexOf(key) === -1) {
          return caches.delete(key)
        }
      }))
    }).then(() => self.clients.claim()) // 当一个 service worker 被初始注册时，页面在下次加载之前不会使用它。 claim() 方法会立即控制这些页面。
  )
})
