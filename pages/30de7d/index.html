docker配置

# nuxt

# .env

NUXT_PORT=9901

# docker-compose.yml

version: '1'
services:
    nuxt_front:
        env_file:
            - ./.env
        container_name: "nuxt_front"
        build:
            context: ./
            dockerfile: ./Dockerfile
            args:
                - NUXT_PORT=${NUXT_PORT}
        ports:
            - "${NUXT_PORT}:${NUXT_PORT}"
        restart: unless-stopped

# Dockerfile

FROM node:18.20 as build

WORKDIR /app

COPY . .

RUN npm install \
    && npm run build

FROM node:18.20 as prod

RUN groupadd -r nuxt && useradd -r -g nuxt nuxt

COPY --from=build /app/.output /app/.output
COPY --from=build /app/.env /app/.env
COPY --from=build /app/package.json /app/package.json
COPY --from=build /app/node_modules /app/node_modules

WORKDIR /app

ENV NODE_ENV=production

ENV NUXT_APP_BASE_URL=/front/

ARG NUXT_PORT

ENV PORT=${NUXT_PORT}

EXPOSE ${NUXT_PORT}

CMD [ "node", "./.output/server/index.mjs" ]

# nest

# .env

_PORT=9902

_DATABASE_HOST=mysql_db
_DATABASE_USERNAME=username
_DATABASE_PASSWORD=password
_DATABASE_NAME=nest

_LINUX_STATIC_DIR=/static

# docker-compose.yml

version: "1"
services:
    nest_web:
        env_file:
            - ./.env
        container_name: "nest_web"
        build:
            context: ./
            dockerfile: ./Dockerfile
            args:
                - _PORT=${_PORT}
        volumes:
            - ${_LINUX_STATIC_DIR}:${_LINUX_STATIC_DIR}
        depends_on:
            - mysql_db
            - redis_db
        ports:
            - "${_PORT}:${_PORT}"
        restart: unless-stopped

    mysql_db:
        container_name: "mysql_db"
        image: mysql:8.0
        ports:
            - "3306:3306"
        volumes:
            - /usr/local/mysql/data:/var/lib/mysql
            - /usr/local/mysql/config:/etc/mysql/config.d
        environment:
            - MYSQL_ROOT_PASSWORD=${_DATABASE_PASSWORD}
            - MYSQL_USER=${_DATABASE_USERNAME}
            - MYSQL_PASSWORD=${_DATABASE_PASSWORD}
            - MYSQL_DATABASE=${_DATABASE_NAME}
            - MYSQL_INITDB_SKIP_TZINFO=no
        restart: unless-stopped
    redis_db:
        container_name: "redis_db"
        image: redis:7.2
        ports:
            - "6379:6379"
        volumes:
            - /usr/local/redis/data:/data
            - /usr/local/redis/config:/usr/local/etc/redis
        command: ["redis-server", "/usr/local/etc/redis/redis.conf"]
        restart: unless-stopped

# Dockerfile

FROM node:18.20 as build

WORKDIR /app

COPY . .

RUN npm install \
    && npm run build

FROM node:18.20 as prod

RUN groupadd -r nest && useradd -r -g nest nest

ENV TZ Asia/Shanghai
RUN ln -snf /usr/share/zoneinfo/$TZ /etc/localtime && echo $TZ > /etc/timezone

COPY --from=build /app/dist /app/dist
COPY --from=build /app/.env /app/.env
COPY --from=build /app/package.json /app/package.json
COPY --from=build /app/node_modules /app/node_modules

WORKDIR /app

ENV NODE_ENV=production

ARG _PORT

EXPOSE ${_PORT}

CMD [ "npm", "run", "start:prod" ]
