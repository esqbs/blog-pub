function add(x: number, y: number = 0, z?: number, ...args: number[]): number {
  return x + (y ?? 0);
}

function add([x, y]: [number, number]) { }

function add(x: number, y: number) {
  return x + y;
}
add.v = 1.1

// 构造签名
let con: { new(x: number, y: number): number }
// 函数重载
function add(x: number, y: number): number;
function add(x: any[], y: any[]): any[];
function add(x: number | any[], y: number | any[]): any {
  if (typeof x === 'number' && typeof y === 'number') {
    return x + y;
  }
  if (Array.isArray(x) && Array.isArray(y)) {
    return [...x, ...y];
  }
}
const a: number = add(1, 2);
const b: number[] = add([1], [2]);

# 9. 接口

interface inter {
  a: number, // 属性签名
  (x: number): number, // 调用签名
  new(x: number): number, // 构造签名

  x?: number, // 可选属性

}

}

// 接口继承

# 10. 类型别名

// 类型别名不会创建新的类型，它只是已有类型的引用
type Point = { x: number, y: number }
// 以下场景允许使用递归类型别名
type T0 = { name: T0 } // 对象类型字面量
type T1 = () => T1 // 函数类型字面量
type T2 = new () => T2 // 构造函数类型字面量
type T3 = Array<T3> // 数组类型
type T4 = [number, T4] // 元组类型
interface A<T> {
  name: T
}
type T5 = A<T5> // 泛型
type json = | string | number | boolean | null | { [prop: string]: json } | json[]; // json

类型别名和接口的区别

  • 类型别名能够表示非对象类型，而接口则只能表示对象类型
  • 接口可以继承其他的接口、类等对象类型，而类型别名则不支持继承
  • 接口具有声明合并的行为，而类型别名则不会进行声明合并

# 11. 类

// 抽象类， 不能实例化, 派生类中必须实行所有抽象成员
abstract class Shape {
  [p: string]: number | (() => number | void | this) // 索引成员
}

interface Color {
  setColor(): void
}

interface Size {
  setSize(): void
}

// 类声明
class Circle extends Shape implements Color, Size { // 单继承, 实现多个接口

  radius: number; // 成员变量
  readonly x: number = 1; // readonly属性，必须在声明或构造函数中初始化
  y: number = 0;

  static version: number = 1; // 静态成员

  // 成员可访问性， 默认类的所有成员都是公有成员
  public a: number = 1;
  protected b: number = 2;
  private c: number = 3;

  // 构造函数
  constructor(public z: number, public readonly w: number) { // 参数成员
    super();
    this.radius = 1;
  }

  area(): number { // 成员函数
    return Math.PI * this.radius * this.radius
  }

  get getY(): number { // get函数
    return this.y;
  }

  set setY(y: number) { // set函数
    this.y = y;
  }

  setColor(): void { }

  setSize(): void { }

  foo(): this { // this类型，表示当前this值的类型
    return this
  }
}

const c: Circle = new Circle(1, 1);  // 类类型

# 二、类型进阶

# 1. 联合类型

// 联合类型字面量
type T0 = string | number;
// 联合类型的类型成员
interface Circle {
  area: bigint
}
interface Rectangle {
  area: number,
  width: number,
  height: number
}
type Shape = Circle | Rectangle
declare const s: Shape
typeof s.area // number | bigint， s下只能保证存在area
// 索引签名
interface T0 {
  [p: string]: string
}
interface T1 {
  [p: string]: number
}
type T = T0 | T1; // 与T0T1等价
interface T0T1 {
  [p: string]: string | number
}
// 调用签名和构造签名同上

# 2. 交叉类型

// 交叉类型字面量
type T0 = string & number; // never
// 交叉类型的类型成员
interface T0 {
  a: string
}
interface T1 {
  b: number
}
type T = T0 | T1; // 与T0T1等价
interface T0T1 {
  a: string
  b: number
}
// 索引签名、调用签名和构造签名同上
// 交叉类型优先级高于联合类型 A & B | C & D 等同于 （A & B） | （C & D）

# 3. 索引类型

// 索引类型查询, 结果类型是了联合类型的 string|number|symbol 子类型
interface Point {
  x: number,
  y: number
}
type T = keyof Point  // 'x'|'y'
type T1 = keyof any // string|number|symbol
type T2 = keyof unknown // never

type A = { a: string, b: boolean }
type B = { c: string, b: boolean }
type T3 = keyof (A | B) // 'b'
type T4 = keyof (A & B) // 'a'|'b'|'c', 相当于 keyof A | keyof B

// 索引类型访问
type T = {
  0: number,
  [prop: string]: number,
}
type T1 = T[0] // number
type T2 = T['1'] // number 

function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key]
}

# 4. 映射对象类型

type K = 'x' | 'y'
type T = number
type T1 = { readonly [P in K]?: T }
// 等价于上面
type T1 = {
    readonly x?: number | undefined;
    readonly y?: number | undefined;
}

type T2 = { [P in 'x']: boolean } // {x: boolean;}
type T3 = { [P in string]: boolean } // {[x:string]: boolean;}

// 将T中所有属性标记为可选
type partial<T> = {
  [P in keyof T]?: T[P]
}
// 将T中所有属性标记为只读
type readonly<T> = {
  readonly [P in keyof T]: T[P]
}
// 修饰符拷贝，默认会拷贝源对象类型的所有属性readonly和?
type pick<T, K extends keyof T> = {
  [P in K]: T[P]
}
// -或+修饰符只作用于?或readonly之前
type require<T> = {
  [P in keyof T]-?: T[P]
}

# 5. 条件类型

type TypeName<T> = T extends string
  ? 'string'
  : T extends number
  ? 'number'
  : T extends boolean
  ? 'boolean'
  : T extends undefined
  ? 'undefined'
  : T extends Function
  ? 'function'
  : 'object';

type T0 = TypeName<'a'>;         // 'string'
type T1 = TypeName<0>;           // 'number'
type T2 = TypeName<true>;        // 'boolean'
type T3 = TypeName<undefined>;   // 'undefined'
type T4 = TypeName<() => void>;  // 'function'
type T5 = TypeName<string[]>;    // 'object'

// 分布式条件类型
type T0<T> = T extends string ? true : false; //裸类型参数，T0<T>是分布式条件类型
type T1<T> = [T] extends [string] ? true : false; // 非裸类型参数，T1<T>不是分布式条件类型
// 分布式条件类型会被展开
T ≡ A | B
T extends U ? X : Y
    ≡ (A extends U ? X : Y) | (B extends U ? X : Y)

// infer, 在extends语句中类型U的位置上允许使用infer关键字来定义可推断的类型变量
type CT<T> = T extends { a: infer M; b: infer N } ? [M, N] : never
type T = CT<{ a: number; b: string }>

# 8. 类型断言

// <T>类型断言
const username = document.getElementById("username")
if (username) {
  (<HTMLInputElement>username).value
}
// as T类型断言
const username = document.getElementById("username")
if (username) {
  (username as HTMLInputElement).value
}

// const断言
function add(x: number, y: number) {
  return x + y;
}
const nums = [1, 2] as const;
//    ~~~~
//    推断出的类型为'readonly [1, 2]'
const total = add(...nums); // 正确

// 非空断言
function isDefined(value: any) {
  return value !== undefined && value !== null
}
function getLength(v: string | undefined) {
  if (!isDefined(v)) {
    return 0
  }
  return v!.length // v可能为undefined
}
